Anyone who uses tmux on their VPS, add this
# creates the tmux session called ssh_tmux if none exists or reattaches to a already existing session with that name. if [[ -z "$TMUX" ]] && [ "$SSH_CONNECTION" != "" ]; then tmux attach-session -t ssh_tmux || tmux new-session -s ssh_tmux fi
