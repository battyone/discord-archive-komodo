#!/usr/bin/env python3 import hashlib import base58 import binascii private_key_WIF = input('key:') first_encode = base58.b58decode(private_key_WIF) private_key_full = binascii.hexlify(first_encode) #print(private_key_full) private_key = private_key_full[2:-8] private_key_static = private_key.decode("utf-8") print('static',private_key_static) extended_key = "bc"+private_key_static print('extended_key', extended_key) first_sha256 = hashlib.sha256(binascii.unhexlify(extended_key)).hexdigest() print('first', first_sha256) second_sha256 = hashlib.sha256(binascii.unhexlify(first_sha256)).hexdigest() print('second', second_sha256) # add checksum to end of extended key final_key = extended_key+second_sha256[:8] # Wallet Import Format = base 58 encoded final_key WIF = base58.b58encode(binascii.unhexlify(final_key)) print (WIF.decode("utf-8"))
