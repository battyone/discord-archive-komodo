`#!/bin/bash
delay=0
listallac=$(cat luke2.json | jq -r .[].ac_name)
echo "$listallac" > listallac

getparamsofac() {

if [ -z "$1" ] || [ "$(cat /home/dwy/script/listallac | grep $1 | wc -l)" -eq 0 ]
then
        echo "Please input AC NAME"
else
        thiskeys=$(cat luke2.json | jq -r ".[] | select(.ac_name==\"$1\")" | jq -r 'keys[]')
        totalparams=""
        coinacname=" -ac_name=$1"
        while read thisk; do
                if [ ! "$thisk" == "ac_name" ]
                then
                        thisparam=$(cat luke2.json | jq -r ".[] | select(.ac_name==\"$1\").$thisk")
                        totalparams+=$(echo " -$thisk=$thisparam")
                fi
        done <<< "$thiskeys"
fi
}

launchac() {
while read line; do
        getparamsofac $line
        echo "komodo-cli $coinacname $totalparams"
        sleep $delay
done <<< "$listallac"
}

getparamsofac RFOX

getparamsofac AAA

getparamsofac

launchac
`
