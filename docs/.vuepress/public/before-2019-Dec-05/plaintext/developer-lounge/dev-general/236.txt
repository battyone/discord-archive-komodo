```#!/bin/bash

komododir=$HOME/komodo/src
list=`cat $komododir/assetchains.json`
for item in $(jq -c 'to_entries[].value' <<<${list}); 
    do
checking=$(jq -rc '.ac_name' <<<${item})
if [[ ${checking} = ${1} ]]; then
        echo "${item}"
fi
done```
