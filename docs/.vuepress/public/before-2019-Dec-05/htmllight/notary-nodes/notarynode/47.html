iguana_arg=${2:-notary} stdbuf -oL $1 ../agents/iguana $iguana_arg & #> iguana.log 2> error.log &

iguana_arg=${2:-notary} stdbuf -oL $1 ../agents/iguana $iguana_arg > iguana.log 2> error.log &

#!/bin/bash komodod &
