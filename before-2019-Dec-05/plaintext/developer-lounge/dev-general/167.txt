        _, v, err := base58.CheckDecode(addr)
        if err != nil {
            return err
        }
        if bytes.Compare([]byte{v}, []byte{0x3C}) != 0 {
            return errors.New("Invalid address. does not have correct prefix")

        return nil
    } else {
        return fmt.Errorf("Invalid length: %d.", len(addr))

}
